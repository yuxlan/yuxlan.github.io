执行上下文

2017-06-07

作业部落链接地址

var color = 'blue';

function changeColor(){
    var anothorColor = "red";

    function swapColor(){
        var tempColor = anothorColor;
        anothorColor = color;
        color = tempColor;
    }

    swapColor();
}

执行上下文可以理解为当前代码的执行环境，会形成一个作用域，JavaScript的运行环境大致包括三种环境：

  1. 全局环境：JavaScript代码运行会首先进入该环境
  2. 函数环境：当函数被调用执行时，会进入当前函数中执行代码
  3. eval
  1. 单线程
  2. 同步执行，只有处于栈顶的上下文处于执行中，其它上下文需要等待
  3. 全局上下文只有唯一一个，在关闭浏览器时出栈
  4. 函数的执行上下文个数没有限制
  5. 每次某个函数被调用，都会有新的上下文为其创建，包括其调用的是自身函数。
function f1(){
    var n = 999;
    function f2(){
        alert(n);
    }
    return f2; // 返回f2，但f2并未被执行
}

var result = f1();

result(); // 相当于执行f2
